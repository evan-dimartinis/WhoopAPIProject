let body = new URLSearchParams({
        grant_type: "refresh_token",
        refresh_token: refreshToken,
        client_id: whoopOAuthConfig.clientID,
        client_secret: whoopOAuthConfig.clientSecret,
        scope: "offline",
      });
      let headers = {
        "Content-Type": "application/x-www-form-urlencoded",
      };
      await fetch(whoopOAuthConfig.tokenURL, {
        body,
        headers,
        method: "POST",
      })
        .then((response) => {
          return response.json();
        })
        .then((ResData) => {
          app.use(session({}));
          const query = new URLSearchParams({
            limit: "1",
          });
          return fetch(
            `https://api.prod.whoop.com/developer/v1/recovery?${query}`,
            {
              headers: {
                Authorization: "Bearer " + ResData.access_token,
              },
              method: "GET",
            }
          );
        })
        .then((response) => {
          return response.json();
        })
        .then((ResData) => {
          console.log("inside (important) fetch: ", ResData);
        })
        .catch((err) => {
          console.log(err);
        });
